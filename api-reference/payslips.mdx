---
title: "Payslip"
description: "Get payslip data"
api: "GET /connect/{account_id}/payslips"
---

## Parameters

<ParamField path="account_id" type="string" required>
  The account_id of the user
</ParamField>

## Response Type

<ResponseField name="payment_status" type="string">
      Status of the payment
    </ResponseField>

    <ResponseField name="type" type="string">
      Frequency of the payment
    </ResponseField>
    
    <ResponseField name="payday" type="string">
      The date payment was made
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      The currency salary was paid in
    </ResponseField>

    <ResponseField name="gross_pay" type="number">
      Total gross pay 
    </ResponseField>
    
    <ResponseField name="amount_paid" type="number">
      The actual amount the user received on the payday
    </ResponseField>
    
    <ResponseField name="net_pay" type="number">
      Total net pay
    </ResponseField>
