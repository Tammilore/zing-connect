---
title: "Transactions"
description: "Get transactions data"
api: "GET /connect/{account_id}/transactions"
---

## Parameters

<ParamField path="account_id" type="string" required>
  The account_id of the user
</ParamField>

<ParamField query="start_date" type="string" placeholder="2023-05-23" required>
  Start date you want transactions from
</ParamField>

<ParamField query="end_date" type="string" placeholder="2023-05-29" required>
  End date you want transactions from
</ParamField>

## Response Fields

<ResponseField name="id" type="string">
      Transaction ID
    </ResponseField>

    <ResponseField name="date" type="string">
      Date of the transaction
    </ResponseField>
    
    <ResponseField name="type" type="string">
      Transaction type, whether credit or debit
    </ResponseField>
    
    <ResponseField name="amount" type="number">
      The transaction amount
    </ResponseField>

    <ResponseField name="balance" type="number">
      Account balance after transaction
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      The currency the transaction was made in
    </ResponseField>
    
    <ResponseField name="narration" type="string">
      Details or description of the transaction
    </ResponseField>
