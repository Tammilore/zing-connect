---
title: "Account Details"
description: "Get account details data"
api: "GET /connect/{account_id}/account"
---

## Parameters

<ParamField path="account_id" type="string" required>
  The account_id of the user
</ParamField>

## Response Fields

<ResponseField name="account" type="Object">
  <Expandable title="properties">
    <ResponseField name="bank" type="Object">
  <Expandable title="properties">
    <ResponseField name="code" type="string">
      The bank code for the account
    </ResponseField>

    <ResponseField name="name" type="string">
      The name of the bank
    </ResponseField>

    <ResponseField name="type" type="string">
      The type of bank
    </ResponseField>
  </Expandable>
</ResponseField>

    <ResponseField name="name" type="string">
      The name of the account holder
    </ResponseField>

    <ResponseField name="type" type="string">
      The type of the bank account connected
    </ResponseField>
    
    <ResponseField name="balance" type="integer">
      User's account balance
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      Currency of the connected account
    </ResponseField>
  
    <ResponseField name="account_number" type="string">
      Account number of the user
    </ResponseField>
  </Expandable>
</ResponseField>
